const express = require('express'); const app = express(); const port = 3000; // Sample menu data (in a real application, this would come from a database) const menu = [ { id: 1, name: 'Pizza', description: 'Delicious pizza with various toppings', price: 10.99 }, { id: 2, name: 'Burger', description: 'Juicy beef burger with lettuce and cheese', price: 8.99 }, { id: 3, name: 'Salad', description: 'Fresh salad with mixed greens and veggies', price: 6.99 } ]; // Serve static files from the 'public' directory app.use(express.static('public')); // Endpoint to get the menu app.get('/menu', (req, res) => { res.json(menu); }); // Start the server app.listen(port, () => { console.log(`Server listening at http://localhost:${port}`); });
